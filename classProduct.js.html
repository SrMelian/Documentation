/**
 * Constructor which create the HTML starting to the object
 * @class Class description
 * @constructor
 * @param {String} amount
 * @param {String} product
 * @param {String} color Hexadecimal color
 */
function Product(amount, product, color) {
	this.amount = amount;
	this.product = product;
	this.color = color;
	this.number = uuid.v1();
	this.element = createInputTag("board", `${this.amount} - ${this.product}`, "checkbox", this.number);
	this.element.style.backgroundColor = color;
	this.element.addEventListener("click", function () {
		fillArrayProductsChecked(this);
	}, false);
}

/**
 * Set the property with the new color, and apply a css style
 * @param {String} color Hexadecimal color
 */
Product.prototype.updateColor = function (color) {
	this.color = color;
	this.element.style.backgroundColor = color;
};

/**
 * Update the basic properties of the product
 * @param {*} amount 
 * @param {*} product 
 */
Product.prototype.editProduct = function (amount, product) {
	this.amount = amount;
	this.product = product;
};
